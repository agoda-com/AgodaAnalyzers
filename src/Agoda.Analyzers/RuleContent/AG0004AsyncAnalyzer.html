﻿

You should not use blocking async code.

Wait,WaitAll,WaitAny gonna block your current thread and prevent your application to use this thread to others works.

Wait()

Noncompliant Code Example

public class TestClass
{
	public void Test()
	{
		var test = new Task(() => { });
		test.Wait();
	}
}

Compliant Code Example

public class TestClass
{
	public async Task Test()
	{
		var test = new Task(() => { });
		var result = await test;		
	}
}

WaitAll()

Noncompliant Code Example

public class TestClass
{
	public void Test()
	{
		var test = new Task(() => { });
		var test2 = new Task(() => { });
		
		Task.WaitAll(test,test2);
	}
}

Compliant Code Example

public class TestClass
{
	public void Test()
	{
		var test = new Task(() => { });
		var test2 = new Task(() => { });
		Task.WhenAll(test,test2);
	}
}
